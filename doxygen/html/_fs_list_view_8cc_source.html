669 void ListView::scrollChange(float x,float y)
670 {
671  //FS_TRACE_WARN("x=%f,y=%f",x,y);
672 
673  float minx=0.0f,maxx=0.0f,miny=0.0f,maxy=0.0f;
674 
675  if(m_contentPanel->getMode()==E_ScrollDirection::VERTICAL)
676  {
677  minx=0;
678  maxx=m_size.x;
679  miny=m_contentPanel->getContentHeight();
680  maxy=0;
681  }
682  else if(m_contentPanel->getMode()==E_ScrollDirection::HORIZONTAL)
683  {
684  minx=0;
685  maxx=m_contentPanel->getContentWidth();
686  miny=-m_size.y;
687  maxy=0;
688 
689  }
690 
691  float middlex=(minx+maxx)/2;
692  float middley=(miny+maxy)/2;
693 
694  switch(m_alignv)
695  {
696  case E_AlignV::TOP:
697  m_contentPanel->setPositionY(y-maxy);
698  break;
699 
700  case E_AlignV::CENTER:
701  m_contentPanel->setPositionY(y-middley);
702  break;
703 
704  case E_AlignV::BOTTOM:
705  m_contentPanel->setPositionY(y-miny);
706  break;
707  default:
708  FS_TRACE_WARN("Unkown Align For Vetical");
709  }
710 
711  switch(m_alignh)
712  {
713  case E_AlignH::LEFT:
714  m_contentPanel->setPositionX(x-minx);
715  break;
716 
717  case E_AlignH::CENTER:
718  m_contentPanel->setPositionX(x-middlex);
719  break;
720 
721  case E_AlignH::RIGHT:
722  m_contentPanel->setPositionX(x-maxx);
723  break;
724 
725  default:
726  FS_TRACE_WARN("Unkown Align For Horizontal");
727  }
