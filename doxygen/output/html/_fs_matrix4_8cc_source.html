55 void Matrix4::mul(const Matrix4& n)
56 {
57  float t00,t01,t02,t03;
58  float t10,t11,t12,t13;
59  float t20,t21,t22,t23;
60  float t30,t31,t32,t33;
61 
62  t00=m00*n.m00+m01*n.m10+m02*n.m20+m03*n.m30;
63  t01=m00*n.m01+m01*n.m11+m02*n.m21+m03*n.m31;
64  t02=m00*n.m02+m01*n.m12+m02*n.m22+m03*n.m32;
65  t03=m00*n.m03+m01*n.m13+m02*n.m23+m03*n.m33;
66 
67  t10=m10*n.m00+m11*n.m10+m12*n.m20+m13*n.m30;
68  t11=m10*n.m01+m11*n.m11+m12*n.m21+m13*n.m31;
69  t12=m10*n.m02+m11*n.m12+m12*n.m22+m13*n.m32;
70  t13=m10*n.m03+m11*n.m13+m12*n.m23+m13*n.m33;
71 
72  t20=m20*n.m00+m21*n.m10+m22*n.m20+m23*n.m30;
73  t21=m20*n.m01+m21*n.m11+m22*n.m21+m23*n.m31;
74  t22=m20*n.m02+m21*n.m12+m22*n.m22+m23*n.m32;
75  t23=m20*n.m03+m21*n.m13+m22*n.m23+m23*n.m33;
76 
77  t30=m30*n.m00+m31*n.m10+m32*n.m20+m33*n.m30;

81

147 void Matrix4::inverse()
148 {
149  float t00,t01,t02,t03;
150  float t10,t11,t12,t13;
151  float t20,t21,t22,t23;
152  float t30,t31,t32,t33;
153 
154  float det=getDetermiant();
155 
156  FS_TRACE_ERROR_ON(det==0,"Matrix Determiant Is Zero");
157  det=1.0f/det;
158 
159  t00=m12*m23*m31 - m13*m22*m31 + m13*m21*m32 - m11*m23*m32 - m12*m21*m33 + m11*m22*m33;
160  t01=m03*m22*m31 - m02*m23*m31 - m03*m21*m32 + m01*m23*m32 + m02*m21*m33 - m01*m22*m33;
161  t02=m02*m13*m31 - m03*m12*m31 + m03*m11*m32 - m01*m13*m32 - m02*m11*m33 + m01*m12*m33;
162  t03=m03*m12*m21 - m02*m13*m21 - m03*m11*m22 + m01*m13*m22 + m02*m11*m23 - m01*m12*m23;
163 
164  t10=m13*m22*m30 - m12*m23*m30 - m13*m20*m32 + m10*m23*m32 + m12*m20*m33 - m10*m22*m33;
165  t11=m02*m23*m30 - m03*m22*m30 + m03*m20*m32 - m00*m23*m32 - m02*m20*m33 + m00*m22*m33;
166  t12=m03*m12*m30 - m02*m13*m30 - m03*m10*m32 + m00*m13*m32 + m02*m10*m33 - m00*m12*m33;
167  t13=m02*m13*m20 - m03*m12*m20 + m03*m10*m22 - m00*m13*m22 - m02*m10*m23 + m00*m12*m23;
168 
169  t20=m11*m23*m30 - m13*m21*m30 + m13*m20*m31 - m10*m23*m31 - m11*m20*m33 + m10*m21*m33;
170  t21=m03*m21*m30 - m01*m23*m30 - m03*m20*m31 + m00*m23*m31 + m01*m20*m33 - m00*m21*m33;
171  t22=m01*m13*m30 - m03*m11*m30 + m03*m10*m31 - m00*m13*m31 - m01*m10*m33 + m00*m11*m33;
172  t23=m03*m11*m20 - m01*m13*m20 - m03*m10*m21 + m00*m13*m21 + m01*m10*m23 - m00*m11*m23;
173 
174  t30=m12*m21*m30 - m11*m22*m30 - m12*m20*m31 + m10*m22*m31 + m11*m20*m32 - m10*m21*m32;
175  t31=m01*m22*m30 - m02*m21*m30 + m02*m20*m31 - m00*m22*m31 - m01*m20*m32 + m00*m21*m32;
176  t32=m02*m11*m30 - m01*m12*m30 - m02*m10*m31 + m00*m12*m31 + m01*m10*m32 - m00*m11*m32;
177  t33=m01*m12*m20 - m02*m11*m20 + m02*m10*m21 - m00*m12*m21 - m01*m10*m22 + m00*m11*m22;
178 
179 
180  set(t00*det,t01*det,t02*det,t03*det,
181  t10*det,t11*det,t12*det,t13*det,
182  t20*det,t21*det,t22*det,t23*det,
183  t30*det,t31*det,t32*det,t33*det);
184 }

193 float Matrix4::getDetermiant()
194 {
195  float r=m03 * m12 * m21 * m30-
196  m02 * m13 * m21 * m30-
197  m03 * m11 * m22 * m30+
198  m01 * m13 * m22 * m30+
199 
200  m02 * m11 * m23 * m30-
201  m01 * m12 * m23 * m30-
202  m03 * m12 * m20 * m31+
203  m02 * m13 * m20 * m31+
204 
205  m03 * m10 * m22 * m31-
206  m00 * m13 * m22 * m31-
207  m02 * m10 * m23 * m31+
208  m00 * m12 * m23 * m31+
209 
210  m03 * m11 * m20 * m32-
211  m01 * m13 * m20 * m32-
212  m03 * m10 * m21 * m32+
213  m00 * m13 * m21 * m32+
214 
215  m01 * m10 * m23 * m32-
216  m00 * m11 * m23 * m32-
217  m02 * m11 * m20 * m33+
218  m01 * m12 * m20 * m33+
219 
220  m02 * m10 * m21 * m33-
221  m00 * m12 * m21 * m33-
222  m01 * m10 * m22 * m33+
223  m00 * m11 * m22 * m33;
224  return r;
225 }

353 {

355  float y = 2 * near / ( top - bottom );
356

359  float c = - ( far + near ) / ( far - near );
360  float d = - 2 * far * near / ( far - near );
361 
362  m00 = x; m01 = 0; m02 = a; m03 = 0;
363  m10 = 0; m11 = y; m12 = b; m13 = 0;

656 void Matrix4::rawSetRotateFromEuler(float rx,float ry,float rz,E_EulerOrientType type)
657 {
658  float cx = Math::cosa( rx ); //a
659  float sx = Math::sina( rx ); //b
660 
661  float cy = Math::cosa( ry ); //c
662  float sy = Math::sina( ry ); //d
663 
664  float cz = Math::cosa( rz ); //e
665  float sz = Math::sina( rz ); //f
666 
667  switch(type)
668  {
669  case E_EulerOrientType::XYZ:
670  m00=cy*cz;
671  m01=-cy*sz;
672  m02=sy;
673 
674  m10=sx*sy*cz+cx*sz;
675  m11=-sx*sy*sz+cx*cz;
676  m12=-sx*cy;
677 
678  m20=-cx*sy*cz+sx*sz;
679  m21=cx*sy*sz+sx*cz;
680  m22=cx*cy;
681  break;
682  case E_EulerOrientType::XZY:
683  m00=cz*cy;
684  m01=-sz;
685  m02=cz*sy;
686 
687  m10=cx*sz*cy+sx*sy;
688  m11=cx*cz;
689  m12=cx*sz*sy-sx*cy;
690 
691  m20=sx*sz*cy-cx*sy;
692  m21=sx*cz;
693  m22=sx*sz*sy+cx*cy;
694  break;
695 
696  case E_EulerOrientType::YXZ:
697  m00=cy*cz+sy*sx*sz;
698  m01=-cy*sz+sy*sx*cz;
699  m02=sy*cx;
700 
701  m10=cx*sz;
702  m11=cx*cz;
703  m12=-sx;
704 
705  m20=-sy*cz+cy*sx*sz;
706  m21=sy*sz+cy*sx*cz;
707  m22=cy*cx;
708  break;
709 
710  case E_EulerOrientType::YZX:
711  m00=cy*cz;
712  m01=-cy*sz*cx+sy*sx;
713  m02=cy*sz*sx+sy*cx;
714 
715  m10=sz;
716  m11=cz*cx;
717  m12=-cz*sx;
718 
719  m20=-sy*cz;
720  m21=sy*sz*cx+cy*sx;
721  m22=-sy*sz*sx+cy*cx;
722  break;
723  case E_EulerOrientType::ZXY:
724  m00=cz*cy-sz*sx*sy;
725  m01=-sz*cx;
726  m02=cz*sy+sz*sx*cy;
727 
728  m10=sz*cy+cz*sx*sy;
729  m11=cz*cx;
730  m12=sz*sy-cz*sx*cy;
731 
732  m20=-cx*sy;
733  m21=sx;
734  m22=cx*cy;
735  break;
736 
737  case E_EulerOrientType::ZYX:
738  m00=cz*cy;
739  m01=-sz*cx+cz*sy*sx;
740  m02=sz*sx+cz*sy*cx;
741 
742  m10=sz*cy;
743  m11=cz*cx+sz*sy*sx;
744  m12=-cz*sx+sz*sy*cx;
745 
746  m20=-sy;
747  m21=cy*sx;
748  m22=cy*cx;
749  break;
750  default:
751  FS_TRACE_ERROR("Unkown EULER Type");
752  }
