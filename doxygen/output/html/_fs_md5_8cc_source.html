66 ** Realiza a rotacao no sentido horario dos bits da variavel 'D' do tipo WORD32.
67 ** Os bits sao deslocados de 'num' posicoes
68 */
69 #define rotate(D, num) (D<<num) | (D>>(WORD-num))

71 /*Macros que definem operacoes relizadas pelo algoritmo md5 */
72 #define F(x, y, z) (((x) & (y)) | ((~(x)) & (z)))
73 #define G(x, y, z) (((x) & (z)) | ((y) & (~(z))))
74 #define H(x, y, z) ((x) ^ (y) ^ (z))
75 #define I(x, y, z) ((y) ^ ((x) | (~(z))))

99 static void word32tobytes (const WORD32 *input, char *output) {
100  int j = 0;
101  while (j<4*4) {
102  WORD32 v = *input++;
103  output[j++] = (char)(v & 0xff); v >>= 8;
104  output[j++] = (char)(v & 0xff); v >>= 8;
105  output[j++] = (char)(v & 0xff); v >>= 8;
106  output[j++] = (char)(v & 0xff);
107  }
108 }

119 /*funcao que implemeta os quatro passos principais do algoritmo MD5 */
120 static void digest(const WORD32 *m, WORD32 *d) {
121  int j;
122  /*MD5 PASSO1 */
123  for (j=0; j<4*4; j+=4) {
124  d[0] = d[0]+ F(d[1], d[2], d[3])+ m[j] + T[j]; d[0]=rotate(d[0], 7);
125  d[0]+=d[1];
126  d[3] = d[3]+ F(d[0], d[1], d[2])+ m[(j)+1] + T[j+1]; d[3]=rotate(d[3], 12);
127  d[3]+=d[0];
128  d[2] = d[2]+ F(d[3], d[0], d[1])+ m[(j)+2] + T[j+2]; d[2]=rotate(d[2], 17);
129  d[2]+=d[3];
130  d[1] = d[1]+ F(d[2], d[3], d[0])+ m[(j)+3] + T[j+3]; d[1]=rotate(d[1], 22);
131  d[1]+=d[2];
132  }
133  /*MD5 PASSO2 */
134  for (j=0; j<4*4; j+=4) {
135  d[0] = d[0]+ G(d[1], d[2], d[3])+ m[(5*j+1)&0x0f] + T[(j-1)+17];
136  d[0] = rotate(d[0],5);
137  d[0]+=d[1];
138  d[3] = d[3]+ G(d[0], d[1], d[2])+ m[((5*(j+1)+1)&0x0f)] + T[(j+0)+17];
139  d[3] = rotate(d[3], 9);
140  d[3]+=d[0];
141  d[2] = d[2]+ G(d[3], d[0], d[1])+ m[((5*(j+2)+1)&0x0f)] + T[(j+1)+17];
142  d[2] = rotate(d[2], 14);
143  d[2]+=d[3];
144  d[1] = d[1]+ G(d[2], d[3], d[0])+ m[((5*(j+3)+1)&0x0f)] + T[(j+2)+17];
145  d[1] = rotate(d[1], 20);
146  d[1]+=d[2];
147  }
148  /*MD5 PASSO3 */
149  for (j=0; j<4*4; j+=4) {
150  d[0] = d[0]+ H(d[1], d[2], d[3])+ m[(3*j+5)&0x0f] + T[(j-1)+33];
151  d[0] = rotate(d[0], 4);
152  d[0]+=d[1];
153  d[3] = d[3]+ H(d[0], d[1], d[2])+ m[(3*(j+1)+5)&0x0f] + T[(j+0)+33];
154  d[3] = rotate(d[3], 11);
155  d[3]+=d[0];
156  d[2] = d[2]+ H(d[3], d[0], d[1])+ m[(3*(j+2)+5)&0x0f] + T[(j+1)+33];
157  d[2] = rotate(d[2], 16);
158  d[2]+=d[3];
159  d[1] = d[1]+ H(d[2], d[3], d[0])+ m[(3*(j+3)+5)&0x0f] + T[(j+2)+33];
160  d[1] = rotate(d[1], 23);
161  d[1]+=d[2];
162  }
163  /*MD5 PASSO4 */
164  for (j=0; j<4*4; j+=4) {
165  d[0] = d[0]+ I(d[1], d[2], d[3])+ m[(7*j)&0x0f] + T[(j-1)+49];
166  d[0] = rotate(d[0], 6);
167  d[0]+=d[1];
168  d[3] = d[3]+ I(d[0], d[1], d[2])+ m[(7*(j+1))&0x0f] + T[(j+0)+49];
169  d[3] = rotate(d[3], 10);
170  d[3]+=d[0];
171  d[2] = d[2]+ I(d[3], d[0], d[1])+ m[(7*(j+2))&0x0f] + T[(j+1)+49];
172  d[2] = rotate(d[2], 15);
173  d[2]+=d[3];
174  d[1] = d[1]+ I(d[2], d[3], d[0])+ m[(7*(j+3))&0x0f] + T[(j+2)+49];
175  d[1] = rotate(d[1], 21);
176  d[1]+=d[2];
177  }
178 }

201 /*
202 ** returned status:
203 * 0 - normal message (full 64 bytes)
204 * 1 - enough room for 0x80, but not for message length (two 4-byte words)
205 * 2 - enough room for 0x80 plus message length (at least 9 bytes free)
206 */
207 static int converte (WORD32 *x, const char *pt, int num, int old_status) {
208  int new_status = 0;
209  char buff[64];
210  if (num<64) {
211  memcpy(buff, pt, num); /* to avoid changing original string */
212  memset(buff+num, 0, 64-num);
213  if (old_status == 0)
214  buff[num] = '\200';
215  new_status = 1;
216  pt = buff;
217  }
218  bytestoword32(x, pt);
219  if (num <= (64 - 9))
220  new_status = 2;
221  return new_status;
222 }

226 void FsMd5_ToBinaryBit(const char *message, long len, char *output) {
227  WORD32 d[4];
228  int status = 0;
229  long i = 0;
230  inic_digest(d);
231  while (status != 2) {
232  WORD32 d_old[4];
233  WORD32 wbuff[16];
234  int numbytes = (len-i >= 64) ? 64 : len-i;
235  /*salva os valores do vetor digest*/
236  d_old[0]=d[0]; d_old[1]=d[1]; d_old[2]=d[2]; d_old[3]=d[3];
237  status = converte(wbuff, message+i, numbytes, status);
238  if (status == 2) put_length(wbuff, len);
239  digest(wbuff, d);
240  d[0]+=d_old[0]; d[1]+=d_old[1]; d[2]+=d_old[2]; d[3]+=d_old[3];
241  i += numbytes;
242  }
243  word32tobytes(d, output);
244 }

247 void FsMd5_ToString(const char* message,long len,char* output)
248 {
249  uint8_t hash_bit[HASHSIZE];
250  FsMd5_ToBinaryBit(message,len,(char*)hash_bit);
251 
252  for(int i=0;i<HASHSIZE;i++)
253  {
254  uint8_t c=hash_bit[i];
255 
256  uint8_t h=c>>4 ;
257  uint8_t l=c&0xf;
258 
259  if (h>=10)
260  {
261  output[i*2]='a'+h-10;
262  }
263  else
264  {
265  output[i*2]='0'+h;
266  }
267 
268  if(l>=10)
269  {
270  output[i*2+1]='a'+l-10;
271  }
272  else
273  {
274  output[i*2+1]='0'+l;
275  }
276  }
277 }
